from sklearn.model_selection import train_test_split
# set de datos con los datos previamente separados para poder analizarlos
X = datos.data
y = datos.target
X_ent , X_pru , y_ent , y_pru =  train_test_split(X ,y , test_size=.3)
from sklearn import svm
from sklearn import metrics
# probando hiperparametros
kernels= ["linear" , "rbf" , "sigmoid"]
gammas = [1, 0.1 , 0.01 , 0.001 , 0.0001 , 0.00001]

for kernel in kernels:
  for gamma in gammas:
# definicion del modelo
    modelo = svm.SVC(kernel = kernel , gamma = gamma)
# entrenamiento
    modelo.fit(X_ent , y_ent)
# predicciones
    predicciones = modelo.predict(X_pru)
# muestreo
    print(f"{kernel}: Gamma: {gamma} EXACTITUD :" , metrics.accuracy_score(y_pru, predicciones))
    print(f"{kernel}: Gamma: {gamma} PRECISION :", metrics.precision_score(y_pru, predicciones))
