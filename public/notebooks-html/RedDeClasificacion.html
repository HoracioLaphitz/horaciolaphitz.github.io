# Importando las bibliotecas necesarias
import tensorflow as tf
import tensorflow_datasets as tfds

# descarga y carga de set de datos "mnist" de tensorflow
# Cargando el conjunto de datos MNIST
datos , metadatos = tfds.load("mnist" , as_supervised=True , with_info=True)
# division de los datos en entrenamiento y prueba
datos_entrenamiento = datos["train"]
datos_prueba = datos["test"]


# Obteniendo los nombres de las clases
nombres_clases = metadatos.features["label"].names
# Definiendo una función para normalizar las imágenes
# transformar de 0-255 -&gt; 0-1 (0=0 , 255=1 , 127=0.5)
# Normalizando los datos de entrenamiento y prueba
def normalizar (imagenes , etiquetas):
    imagenes = tf.cast(imagenes , tf.float32)
    imagenes = imagenes/255
    return imagenes , etiquetas
datos_entrenamiento = datos_entrenamiento.map(normalizar)
datos_prueba = datos_prueba.map(normalizar)

# Agregando los datos en caché para mejorar el rendimiento

datos_entrenamiento= datos_entrenamiento.cache()
datos_prueba= datos_prueba.cache()

# Creando el modelo de red neuronal
# tf.keras.Sequential -&gt; para especificar las capas

modelo = tf.keras.Sequential([
    # 748 datos de entrada -&gt; imagenes estiradas para entrenar de mejor manera el modelo
    tf.keras.layers.Flatten(input_shape=(28,28,1)), # 1 = blanco y negro
    # capa oculta
    tf.keras.layers.Dense(50, activation=tf.nn.relu),
    tf.keras.layers.Dense(50,activation=tf.nn.relu),
    # capa de salida
    tf.keras.layers.Dense(10, activation=tf.nn.softmax),
])
# compilando el modelo
# Compilando el modelo con el optimizador Adam y la función de pérdida de entropía cruzada categórica
modelo.compile(
    optimizer = "adam",
    # le permite al modelo ajustarse de manera adecuada cuando se hace una clasificacion en lugar de una regresion
    loss=tf.keras.losses.SparseCategoricalCrossentropy(),
    metrics=["accuracy"]

)
# pre procesamiento adicional
# Definiendo el tamaño del lote
TAMAÑO_LOTE = 32
datos_entrenamiento = datos_entrenamiento.repeat().shuffle(60000).batch(TAMAÑO_LOTE)
datos_prueba = datos_prueba.batch(TAMAÑO_LOTE)
import math

# entrenamiento

entrenamiento = modelo.fit(
    # X
    datos_entrenamiento, epochs=10 ,
    # PASO PARA QUE SEPA BIEN COMO TRABAJAR CON LOTES
    steps_per_epoch=math.ceil(60000/TAMAÑO_LOTE)
)

# Funciones para graficar las imágenes y las predicciones
def graficar_imagen(i, arr_predicciones, etiquetas_reales, imagenes):
  arr_predicciones, etiqueta_real, img = arr_predicciones[i], etiquetas_reales[i], imagenes[i]
  plt.grid(False)
  plt.xticks([])
  plt.yticks([])

  plt.imshow(img[...,0], cmap=plt.cm.binary)

  etiqueta_prediccion = np.argmax(arr_predicciones)
  if etiqueta_prediccion == etiqueta_real:
    color = 'blue'
  else:
    color = 'red'

  plt.xlabel("{} {:2.0f}% ({})".format(nombres_clases[etiqueta_prediccion],
                                100*np.max(arr_predicciones),
                                nombres_clases[etiqueta_real]),
                                color=color)

def graficar_valor_arreglo(i, arr_predicciones, etiqueta_real):
  arr_predicciones, etiqueta_real = arr_predicciones[i], etiqueta_real[i]
  plt.grid(False)
  plt.xticks([])
  plt.yticks([])
  grafica = plt.bar(range(10), arr_predicciones, color="#777777")
  plt.ylim([0, 1])
  etiqueta_prediccion = np.argmax(arr_predicciones)

  grafica[etiqueta_prediccion].set_color('red')
  grafica[etiqueta_real].set_color('blue')
