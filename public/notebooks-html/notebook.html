Generación de Datos

# Generar datos de ejemplo
np.random.seed(42)
n_samples = 100  # Usar slider cuando Mercury esté activo

data = pd.DataFrame({
    'x': np.random.randn(n_samples),
    'y': np.random.randn(n_samples),
    'categoria': np.random.choice(['A', 'B', 'C'], n_samples)
})

print(f"Dataset generado con {len(data)} filas")
data.head()

Visualización

plt.figure(figsize=(10, 6))
sns.scatterplot(data=data, x='x', y='y', hue='categoria', s=100, alpha=0.6)
plt.title('Visualización Interactiva de Datos')
plt.xlabel('Variable X')
plt.ylabel('Variable Y')
plt.legend(title='Categoría')
plt.tight_layout()
plt.show()

Estadísticas Descriptivas

print("Estadísticas por categoría:\n")
