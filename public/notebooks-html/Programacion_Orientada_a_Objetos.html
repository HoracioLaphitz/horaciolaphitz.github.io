class Personaje:
    def __init__(self, nombre, fuerza, inteligencia, defensa, vida):
        self.nombre = nombre
        self.__fuerza = fuerza
        self.__inteligencia = inteligencia
        self.__defensa = defensa
        self.__vida = vida
        self.__aguante = fuerza * vida
        self.__turno = False

    def atributos(self):
        print(self.nombre, ":", sep="")
        print("-Fuerza : ", self.__fuerza)
        print("-Inteligencia : ", self.__inteligencia)
        print("-Defensa : ", self.__defensa)
        print("-Vida: ", self.__vida)

    def subir_nivel(self, fuerza, inteligencia, defensa):
        self.__fuerza = self.__fuerza + fuerza
        self.__inteligencia = self.__inteligencia + inteligencia
        self.__defensa = self.__defensa + defensa

    def esta_vivo(self):
        return self.vida &gt; 0

    def morir(self):
        self.vida = 0
        print(self.nombre, "ha muerto")

    # mecanismo por el que el personaje hace daño

    def daño(self, enemigo):
        return self.__fuerza - enemigo.defensa

    def atacar(self, enemigo):
        daño = self.daño(enemigo)
        enemigo.vida = enemigo.__vida - daño
        print(
            self.nombre,
            "hizo la cantida de ",
            daño,
            "puntos de daño a ",
            enemigo.nombre,
        )
        if enemigo.esta_vivo():
            print("La vida de ", enemigo.nombre, "es", enemigo.vida)
        else:
            enemigo.morir()
